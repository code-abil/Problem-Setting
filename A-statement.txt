One day, Vasya and Petya found themselves on a mysterious island. There are n coins various values on the island.
Since, both of them are greedy, they decided to pick up coins and be rich. They pick up coins according to the following algorithm.
- Vasya picks up coin first.
- They take turns alternatively.
- Both of them can only pick up one coin at a time.
- They stop picking up the coins, when there is nothing left.
Chef wants to know who gained more money in total. Can you help him?


Input:
- The first line of input contains a single number n.
- The second line of input contains n space seperated integers c1, c2, ..., cn.


Output:
- If Vasya gets more money, print "Vasya".
- Else if Petya gets more money, print "Petya".
- Else, print "Tie".


Constraints:
- 1 <= n <= 10^5
- 1 <= c[i] <= 10^3 for each valid i


Example Input:
3
1 4 10


Example Output:
Vasya


Sample Explanation:
Vasya picks up coin with value 10.
Then, Petya picks up coin with value 4.
Vasya picks up coin with value 1.
Total money with Vasya = 10 + 1 = 11.
Total money with Petya = 4.
Hence, Vasya got more money in total.

template <typename A, typename B>
string to_string(pair<A, B> p) {
  return "(" + to_string(p.first) + ", " + to_string(p.second) + ")";
}

template <typename A>
string to_string(A v) {
  bool first = true;
  string res = "{";
  for (const auto &x : v) {
    if (!first) {
      res += ", ";
    }
    first = false;
    res += to_string(x);
  }
  res += "}";
  return res;
}

void test_case() {
  int n;
  cin >> n;
  vector<int> C(n + 1);
  for (int i = 1; i <= n; ++i) {
    cin >> C[i];
  }
  sort(C.begin() + 1, C.end());
  int vasya = 0, petya = 0;
  for (int i = n, done = 0; i >= 1; --i, ++done) {
    if (done & 1) {
      vasya += C[i];
    } else {
      petya += C[i];
    }
  }
  if (vasya > petya) {
    cout << "Vasya\n";
  } else if (petya > vasya) {
    cout << "Petya\n";
  } else {
    cout << "Tie\n";
  }
}
