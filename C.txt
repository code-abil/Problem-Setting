Problem Statement:
You are given a sequence of length n, A = a1, a2, ..., an. 
You get a new sequence B of n numbers from A by the following relation:
B[i] = Product of all numbers of A except A[i]
Let C[i] = Number of unique factors of B[i]
You are asked to print the sum of all elements of C modulo 10^9 + 7.


Input:
- The first line of input contains a single integer n.
- The second line of input contains n space separated integers a1, a2,..., an.


Output:
Print a single integer - the sum of all elements of C modulo 10^9 + 7.


Constraints:
- 2 <= n <= 10^5
- 1 <= A[i] <= 10^5 for each valid i


Example Input:
3
5 6 4


Example Output:
22


Sample Explanation:
Array B: {24, 20, 30}
24 has 8 factors: {1, 2, 3, 4, 6, 8, 12, 24}
20 has 6 factors: {1, 2, 4, 5, 10, 20}
30 has 8 factors: {1, 2, 3, 5, 6, 10, 15, 30}
So, Array C: {8, 6, 8}
Hence, the answer is = 8 + 6 + 8 = 22

template <uint32_t mod>
class Modular {
private:
  uint32_t n;

public:
  Modular(int64_t _n = 0) : n((_n >= 0 ? _n : mod - (-_n) % mod) % mod) {}

  uint32_t get() const {
    return n;
  }

  bool operator ==(const Modular& o) const {
    return n == o.n;
  }

  bool operator !=(const Modular& o) const {
    return n != o.n;
  }

  Modular& operator +=(const Modular& o) {
    n += o.n;
    n = (n < mod ? n : n - mod);
    return *this; 
  }

  Modular& operator -=(const Modular& o) {
    n += mod - o.n;
    n = (n < mod ? n : n - mod);
    return *this;
  }

  Modular& operator *=(const Modular& o) {
    n = uint64_t(n) * o.n % mod;
    return *this;
  }

  Modular& operator /=(const Modular& o) {
    return (*this) *= o.inv();
  }

  Modular operator +(const Modular& o) const {
    return Modular(*this) += o;
  }

  Modular operator -(const Modular& o) const {
    return Modular(*this) -= o;
  }

  Modular operator *(const Modular& o) const {
    return Modular(*this) *= o;
  }

  Modular operator /(const Modular& o) const {
    return Modular(*this) /= o;
  }

  Modular pow(uint64_t b) const {
    Modular ans(1), m = Modular(*this);
    while (b) {
      if (b & 1) {
        ans *= m;
      }
      m *= m;
      b >>= 1;
    }
    return ans;
  }

  Modular inv() const {
    int32_t a = n, b = mod, u = 0, v = 1;
    while (a) {
      int32_t t = b / a;
      b -= t * a;
      swap(a, b);
      u -= t * v;
      swap(u, v);
    }
    assert(b == 1);
    return Modular(u);
  }
};
